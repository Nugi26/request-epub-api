// search books using google book api
query {
  searchbook(keywords: "graphql") {
    totalItems
    items {
      gbook_id
      title
      subtitle
      authors
      publisher
      published_date
      description
      page_count
      maturity_rating
      small_thumbnail
      thumbnail
      average_rating
      ratings_count
      requestedId
    }
  }
}
